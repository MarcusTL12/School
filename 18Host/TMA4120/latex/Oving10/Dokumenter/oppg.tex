\subsection*{18}


\begin{gather*}
	a_n = n^2 \left(\frac{i}{4}\right)^n,
	\qquad
	\abs{\frac{a_{n + 1}}{a_n}} = \frac{(n + 1)^2}{4 n^2}
	\\
	\lim_{n \rightarrow \infty}{\frac{(n + 1)^2}{4 n^2}} = \frac{1}{4} < \infty \Rightarrow \text{Series is convergent}
\end{gather*}
